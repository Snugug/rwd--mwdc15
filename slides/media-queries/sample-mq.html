Fluid Changing Custom Grid

$medium: 1 + 2 + 1/6; // Two column custom grid
$big: 1 + 3 + 1 + 1/3 * 2; // Three column custom grid


.content {
  @media (min-width: 600px) {
    float: right;
    width: percentage(2 / $medium);
  }

  @media (min-width: 900px) {
    float: left;
    width: percentage(3 / $big);
    margin-left: percentage((1 + 1/3) / $big);
  }
}
.sidebar {
  @media (min-width: 600px) {
    float: left;
    width: percentage(1 / $medium);
  }

  @media (min-width: 900px) {
    float: right;
    width: percentage(1 / $big);
  }
}
